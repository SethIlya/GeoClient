# docker-compose.yml
services:
  db:
    image: postgis/postgis:14-3.3
    volumes:
      - postgres_data:/var/lib/postgresql/data/
    env_file:
      - ./.env
    restart: unless-stopped

  web:
    build: .
    volumes:
      # --- ИЗМЕНЕНИЕ: Заменяем bind mount на именованный том ---
      - mediafiles:/app/mediafiles
      # - staticfiles:/app/staticfiles  # Также можно сделать и для статики, если нужно
    ports:
      - "8000:8000"
    env_file:
      - ./.env
    depends_on:
      - db
    restart: unless-stopped

# --- ИЗМЕНЕНИЕ: Указываем, что это именованные тома ---
volumes:
  postgres_data:
  mediafiles:
